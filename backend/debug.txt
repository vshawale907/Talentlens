32: // ─── JSON Parser with fallback ─────────────────────────────────────────────
33: const sanitizeAndValidateJSON = <T>(raw: string): T => {
34:     try {
35:         let jsonStr = raw.trim();
36:         const match = raw.match(/```(?:json)?\s*([\s\S]*?)\s*```/i);
37:         if (match) {
38:             jsonStr = match[1].trim();
39:         } else {
40:             const firstBrace = jsonStr.indexOf('{');
41:             const lastBrace = jsonStr.lastIndexOf('}');
42:             const firstBracket = jsonStr.indexOf('[');
43:             const lastBracket = jsonStr.lastIndexOf(']');
44: 
45:             let start = -1;
46:             let end = -1;
47: 
48:             if (firstBrace !== -1 && (firstBracket === -1 || firstBrace < firstBracket)) {
49:                 start = firstBrace;
50:                 end = lastBrace;
51:             } else if (firstBracket !== -1) {
52:                 start = firstBracket;
53:                 end = lastBracket;
54:             }
55:             if (start !== -1 && end !== -1 && end > start) {
56:                 jsonStr = jsonStr.substring(start, end + 1);
57:             }
58:         }
59:         return JSON.parse(jsonStr) as T;
60:     } catch (err) {
61:         throw new AppError('JSON_PARSE_FAILED', 500, 'AI_PARSE_ERROR');
62:     }
63: };

67: // Core Retry wrapper for provider models
68: const attemptSingleGeneration = async <T>(
69:     genFn: (prompt: string) => Promise<string>,
70:     userPrompt: string,
71:     provider: string,
72:     model: string
73: ): Promise<T> => {
74:     let attempt = 0;
75:     while (attempt < 2) {
76:         attempt++;
77:         const requestStart = Date.now();
78:         try {
79:             const raw = await genFn(userPrompt);
80:             const latency = Date.now() - requestStart;
81:             try {
82:                 const parsed = sanitizeAndValidateJSON<T>(raw);
83:                 logger.info(`[AI] ${provider} (${model}) SUCCESS. Latency: ${latency}ms`);
84:                 return parsed;
85:             } catch (parseErr) {
86:                 logger.warn(`[AI] ${provider} (${model}) Parse Failed: Invalid JSON formatting. Retrying with stricter prompt...`);
87:                 userPrompt += '\n\nCRITICAL FIX: Your last response was invalid JSON. Return ONLY strict, valid JSON matching the schema.';
88:             }
89:         } catch (apiErr: any) {
90:             const latency = Date.now() - requestStart;
91:             const msg = apiErr.message || '';
92:             
93:             if (msg.includes('429') && !msg.includes('limit: 0') && !msg.includes('quota')) {
94:                 if (attempt === 1) {
95:                     logger.warn(`[AI] ${provider} (${model}) Rate Limit hit. Retrying in 2s...`);
96:                     await delay(2000);
97:                     continue;
98:                 }
99:             }
100:             logger.warn(`[AI] ${provider} (${model}) FAILED. Reason: ${msg.slice(0, 100)}. Latency: ${latency}ms`);
101:             throw apiErr; 
102:         }
103:     }
104:     throw new Error('JSON parsing failed after retry');
105: };

107: // ─── Core LLM caller ──────────────────────────────────────────────────────
108: const callGPT = async <T>(systemPrompt: string, userPrompt: string, maxTokens = 3000): Promise<T> => {
109: 
110:     if (genAI) {
111:         const geminiModels = ['gemini-2.0-flash', 'gemini-1.5-flash', 'gemini-pro'];
112:         for (const modelName of geminiModels) {
113:             try {
114:                 return await attemptSingleGeneration<T>(
115:                     async (uPrompt) => {
116:                         const model = genAI.getGenerativeModel({ model: modelName });
117:                         const result = await model.generateContent({
118:                             contents: [{ role: 'user', parts: [{ text: systemPrompt + '\n\n' + uPrompt }] }],
119:                             generationConfig: { temperature: 0.3, maxOutputTokens: maxTokens }
120:                         });
121:                         return result.response.text();
122:                     },
123:                     userPrompt, 'Gemini', modelName
124:                 );
125:             } catch (e: any) {
126:                 continue;
127:             }
128:         }
129:     }
130: 
131:     if (groq) {
132:         const groqModels = ['llama-3.3-70b-versatile'];
133:         for (const modelName of groqModels) {
134:             try {
135:                return await attemptSingleGeneration<T>(
136:                     async (uPrompt) => {
137:                         const response = await groq.chat.completions.create({
138:                             model: modelName,
139:                             messages: [
140:                                 { role: 'system', content: systemPrompt },
141:                                 { role: 'user', content: uPrompt },
142:                             ],
143:                             temperature: 0.3, max_tokens: maxTokens, response_format: { type: 'json_object' }
144:                         });
145:                         return response.choices[0]?.message?.content || '{}';
146:                     },
147:                     userPrompt, 'Groq', modelName
148:                );
149:             } catch (e: any) {
150:                continue;
151:             }
152:         }
153:     }
154: 
155:     if (openai) {
156:         const gptModels = [config.OPENAI_MODEL || 'gpt-4o'];
157:         for (const modelName of gptModels) {
158:             try {
159:                return await attemptSingleGeneration<T>(
160:                     async (uPrompt) => {
161:                         const response = await openai.chat.completions.create({
162:                             model: modelName,
163:                             messages: [
164:                                 { role: 'system', content: systemPrompt },
165:                                 { role: 'user', content: uPrompt },

167:                             temperature: 0.3, max_tokens: maxTokens, response_format: { type: 'json_object' }

170:                     },
171:                     userPrompt, 'OpenAI', modelName
172:                );
173:             } catch (e: any) {
174:                continue;
175:             }
176:         }
